Dictionary Methods in Python

keys()


student = {"name": "Harsha", "age": 20}
print(student.keys())
dict_keys(['name', 'age'])

get()


print(student.get("name"))
print(student.get("city"))
Harsha
None

update()


student.update({"city": "Hyderabad"})
print(student)
{'name': 'Harsha', 'age': 20, 'city': 'Hyderabad'}

pop()


student.pop("age")
print(student)
{'name': 'Harsha', 'city': 'Hyderabad'}

clear()


student.clear()
print(student)
{}
